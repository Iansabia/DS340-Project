def create_sequences(
    X: np.ndarray,           # shape (n_rows, n_features), pre-scaled
    y: np.ndarray,           # shape (n_rows,) — target per row
    lookback: int,           # e.g. 6
    group_ids: np.ndarray,   # shape (n_rows,) — pair_id per row, used to prevent windows crossing pair boundaries
) -> tuple[np.ndarray, np.ndarray]:
    """Return (X_seq, y_seq) where X_seq shape = (n_windows, lookback, n_features) and y_seq shape = (n_windows,).
    A window at row i predicts y[i] using X[i-lookback+1 : i+1] — i.e. the target is the NEXT spread-change aligned with the last input row.
    Windows that would cross group boundaries are dropped.
    Pairs with fewer than lookback rows contribute zero windows."""

class EarlyStopping:
    """Tracks best val loss, returns True from .should_stop() once patience exceeded."""
    def __init__(self, patience: int = 10, min_delta: float = 1e-4) -> None: ...
    def step(self, val_loss: float) -> bool:
        """Return True if training should stop. Updates internal best_loss."""

def set_seed(seed: int) -> None:
    """Seed numpy, torch, torch.cuda (if available). Also sets torch.backends.cudnn.deterministic=True."""

def get_device() -> "torch.device":
    """Return torch.device('cuda') if torch.cuda.is_available() else torch.device('cpu')."""

def fit_feature_scaler(
    X: pd.DataFrame,
    bool_cols: list[str],  # columns cast to float {0.0, 1.0} before fit
) -> "sklearn.preprocessing.StandardScaler":
    """Cast bool_cols to float in-place (copy), REJECT any zero-variance columns with a ValueError
    listing the offending column names, then fit StandardScaler on all columns and return fitted scaler."""

def apply_feature_scaler(
    X: pd.DataFrame,
    scaler: "sklearn.preprocessing.StandardScaler",
    bool_cols: list[str],
) -> np.ndarray:
    """Cast bool_cols to float (copy), transform with scaler, return ndarray (n_rows, n_features)."""

    Create `src/models/sequence_utils.py` implementing exactly these signatures (copy verbatim from the `<interfaces>` block above):

    1. `create_sequences(X, y, lookback, group_ids) -> tuple[np.ndarray, np.ndarray]`:
       - Convert X to np.ndarray if pd.DataFrame
       - Loop over unique group_ids (preserving first-occurrence order)
       - For each group, collect contiguous row indices; skip groups with fewer rows than lookback
       - For each valid window end index i in [lookback-1, len(group)-1], stack X[i-lookback+1:i+1] as a sequence, append y[i] as target
       - Stack all windows into X_seq (shape: n_windows, lookback, n_features) and y_seq (shape: n_windows,)
       - Return both as np.ndarray dtype=float

    2. `class EarlyStopping`:
       - `__init__(self, patience: int = 10, min_delta: float = 1e-4)`
       - Internal state: `best_loss = float('inf')`, `counter = 0`
       - `step(val_loss) -> bool`: if val_loss < best_loss - min_delta: best_loss=val_loss, counter=0, return False. Else counter+=1; return counter >= patience.

    3. `set_seed(seed)`:
       - `np.random.seed(seed)`
       - `torch.manual_seed(seed)`
       - If `torch.cuda.is_available()`: `torch.cuda.manual_seed_all(seed)`
       - `torch.backends.cudnn.deterministic = True`
       - `torch.backends.cudnn.benchmark = False`

    4. `get_device() -> torch.device`:
       - `return torch.device("cuda" if torch.cuda.is_available() else "cpu")`

    5. `fit_feature_scaler(X, bool_cols) -> StandardScaler`:
       - `X = X.copy()`; cast each col in bool_cols to float
       - **Zero-variance safety guard (MANDATORY):** compute `std = X.std(axis=0).to_numpy()`. If `(std == 0).any()`, identify the offending column names via `bad_cols = X.columns[std == 0].tolist()` and raise `ValueError(f"fit_feature_scaler: zero-variance columns detected (would produce NaN after scaling): {bad_cols}")`. This surfaces upstream feature-engineering bugs loudly instead of silently poisoning training with NaNs.
       - `scaler = StandardScaler(); scaler.fit(X.values)`; return scaler

    6. `apply_feature_scaler(X, scaler, bool_cols) -> np.ndarray`:
       - `X = X.copy()`; cast each col in bool_cols to float
       - `return scaler.transform(X.values)`
