    **scripts/oracle_keepalive.sh:**
    Oracle reclaims idle free-tier VMs after ~7 days of low CPU usage.
    A brief CPU burst every 10 minutes prevents reclamation:
    ```bash
    #!/bin/bash
    # CPU keep-alive: brief burst to prevent Oracle free-tier reclamation
    dd if=/dev/urandom bs=1M count=10 | md5sum > /dev/null 2>&1
    echo "$(date -u '+%Y-%m-%dT%H:%M:%SZ'): keepalive ping"
    ```

    **scripts/oracle_autocommit.sh:**
    ```bash
    #!/bin/bash
    # Hourly auto-commit of trading results to GitHub
    set -euo pipefail
    cd "$(dirname "$0")/.."
    
    # Only commit if there are changes in data/live/
    if git diff --quiet data/live/ 2>/dev/null && git diff --cached --quiet data/live/ 2>/dev/null; then
        echo "$(date -u): no changes to commit"
        exit 0
    fi

    # Stage live data files (ignore errors for missing files)
    git add data/live/bars.parquet \
            data/live/positions.db \
            data/live/position_history.jsonl \
            data/live/pair_classifications.json \
            2>/dev/null || true

    # Count open positions for commit message
    OPEN_POS=$(python3 -c "
    import sqlite3, os
    db = 'data/live/positions.db'
    if os.path.exists(db):
        conn = sqlite3.connect(db)
        count = conn.execute('SELECT COUNT(*) FROM positions').fetchone()[0]
        conn.close()
        print(count)
    else:
        print(0)
    " 2>/dev/null || echo "?")

    TIMESTAMP=$(date -u '+%Y-%m-%dT%H:%M:%SZ')
    git commit -m "auto: trading results ${TIMESTAMP} (${OPEN_POS} open positions)" || true
    
    # Push with one retry
    git push || (sleep 5 && git push) || echo "$(date -u): push failed, will retry next hour"
    
    echo "$(date -u): committed and pushed (${OPEN_POS} open positions)"
    ```

    Make all three scripts executable: chmod +x scripts/*.sh
