<!-- SQLite schema for positions: -->
```sql
CREATE TABLE IF NOT EXISTS positions (
    pair_id TEXT PRIMARY KEY,
    kalshi_ticker TEXT NOT NULL,
    direction TEXT NOT NULL,        -- "short_spread" or "long_spread"
    entry_spread REAL NOT NULL,
    entry_time TEXT NOT NULL,       -- ISO 8601 UTC
    entry_kalshi_price REAL NOT NULL,
    entry_poly_price REAL NOT NULL,
    current_spread REAL NOT NULL,
    bars_held INTEGER NOT NULL DEFAULT 0,
    max_spread_since_entry REAL NOT NULL,
    min_spread_since_entry REAL NOT NULL,
    unrealized_pnl REAL NOT NULL DEFAULT 0.0,
    tier TEXT NOT NULL,             -- DAILY/WEEKLY/MONTHLY/QUARTERLY
    bar_interval_seconds INTEGER NOT NULL,
    resolution_date TEXT,           -- ISO 8601 or NULL
    spread_history TEXT NOT NULL DEFAULT '[]'  -- JSON array of last N spreads
);

CREATE TABLE IF NOT EXISTS closed_positions (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    pair_id TEXT NOT NULL,
    kalshi_ticker TEXT NOT NULL,
    direction TEXT NOT NULL,
    entry_spread REAL NOT NULL,
    entry_time TEXT NOT NULL,
    exit_time TEXT NOT NULL,
    entry_kalshi_price REAL NOT NULL,
    entry_poly_price REAL NOT NULL,
    exit_spread REAL NOT NULL,
    bars_held INTEGER NOT NULL,
    realized_pnl REAL NOT NULL,
    exit_reason TEXT NOT NULL,
    tier TEXT NOT NULL,
    max_spread REAL NOT NULL,
    min_spread REAL NOT NULL
);
```

<!-- 5 exit rules with concrete thresholds from planning guidance: -->
1. Take Profit:           spread narrowed > 50% from entry
                          short_spread: current_spread < entry_spread * 0.5
                          Example: entered at 50pp, exit at 25pp or below
2. Stop Loss:             spread widened > 30% beyond entry
                          short_spread: current_spread > entry_spread * 1.3
                          Example: entered at 50pp, exit if hits 65pp
3. Time Stop:             max hold by tier (in bars):
                          DAILY: 4 bars, WEEKLY: 24 bars, MONTHLY: 168 bars, QUARTERLY: 500 bars
4. Momentum Exit:         3 consecutive bars moving against position
                          short_spread: 3 consecutive bars where spread increased
5. Resolution Proximity:  < 1 day (24 hours) to resolution -> exit

Priority order: RESOLUTION > STOP_LOSS > TAKE_PROFIT > MOMENTUM > TIME_STOP
(Resolution and stop-loss are safety exits, highest priority)

<!-- ExitReason enum values: -->
TAKE_PROFIT, STOP_LOSS, MOMENTUM, TIME_STOP, RESOLUTION_EXIT, MANUAL

<!-- P&L computation (from backtester.py direction-aware logic): -->
short_spread: realized_pnl = entry_spread - exit_spread  (profit when spread narrows)
long_spread:  realized_pnl = exit_spread - entry_spread  (profit when spread widens)
```

    **src/live/position_manager.py (exit rule methods):**
    - MAX_BARS = {"DAILY": 4, "WEEKLY": 24, "MONTHLY": 168, "QUARTERLY": 500, "UNKNOWN": 168}
    - _check_take_profit(pos) -> bool:
      if direction == "short_spread": current_spread < entry_spread * 0.5
      if direction == "long_spread": abs(current_spread) < abs(entry_spread) * 0.5
    - _check_stop_loss(pos) -> bool:
      if direction == "short_spread": current_spread > entry_spread * 1.3
      if direction == "long_spread": current_spread < entry_spread * 1.3 (spread moved further negative)
    - _check_time_stop(pos) -> bool: pos.bars_held >= MAX_BARS.get(pos.tier, 168)
    - _check_momentum(pos) -> bool:
      Parse spread_history JSON, need at least 4 entries (to see 3 changes)
      For short_spread: check last 3 diffs are all positive (spread increasing = against us)
      For long_spread: check last 3 diffs are all negative (spread decreasing = against us)
    - _check_resolution(pos, now) -> bool:
      If resolution_date is None: return False
      Parse resolution_date ISO string, compute hours remaining
      Return hours_remaining < 24
    - check_exits(pair_id, now) -> ExitReason | None:
      Load position from SQLite
      Apply rules in priority order: RESOLUTION > STOP_LOSS > TAKE_PROFIT > MOMENTUM > TIME_STOP
      Return first triggered ExitReason, or None
    - check_all_exits(now) -> list[tuple[str, ExitReason]]:
      Get all open positions, call check_exits for each, return list of triggered (pair_id, reason)
